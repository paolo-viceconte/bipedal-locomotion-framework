33 void ContactPhaseList::createPhases()
34 {
35  m_phases.clear();
36 
37  std::map<double, std::unordered_map<std::string, ContactList::const_iterator>> activations,
38  deactivations;
39 
40  for (ContactListMap::iterator list = m_contactLists.begin(); list != m_contactLists.end();
41  ++list)
42  {
43  const std::string& key = list->first;
44  for (ContactList::const_iterator step = list->second.begin(); step != list->second.end();
45  ++step)
46  {
47  activations[step->activationTime][key] = step; // By construction, there is only a step
48  // given a key and activationTime
49  deactivations[step->deactivationTime][key] = step;
50  }
51  }
52 
53  if (!activations.size())
54  {
55  return;
56  }
57 
58  ContactPhase currentPhase;
59  currentPhase.beginTime = activations.begin()->first;
60  currentPhase.activeContacts = activations.begin()->second;
61 
62  activations.erase(activations.begin());
63 
64  while ((activations.size() + deactivations.size()) > 1)
65  {
66  if ((activations.size() == 0)
67  || (deactivations.begin()->first <= activations.begin()->first))
68  {
69  // Here I need to remove from the current phase the contacts that are going to end
70  currentPhase.endTime = deactivations.begin()->first;
71  m_phases.push_back(currentPhase);
72 
73  currentPhase.beginTime = deactivations.begin()->first;
74 
75  for (auto& toBeRemoved : deactivations.begin()->second)
76  {
77  currentPhase.activeContacts.erase(toBeRemoved.first); // Erase all the contacts
78  // which are deactivativated
79  // in this instant
80  }
81 
82  deactivations.erase(deactivations.begin());
83 
84  if (activations.size() && (deactivations.begin()->first == activations.begin()->first))
85  {
86  currentPhase.activeContacts.insert(activations.begin()->second.begin(),
87  activations.begin()->second.end()); // Add the
88  // new
89  // contacts
90  // to the
91  // list.
92 
93  activations.erase(activations.begin());
94  }
95  } else // (activations.begin()->first < deactivations.begin()->first)
96  {
97  currentPhase.endTime = activations.begin()->first;
98  m_phases.push_back(currentPhase);
99 
100  currentPhase.beginTime = activations.begin()->first;
101  currentPhase.activeContacts.insert(activations.begin()->second.begin(),
102  activations.begin()->second.end()); // Add the new
103  // contacts to
104  // the list.
105 
106  activations.erase(activations.begin());
107  }
108  }
109 
110  assert(deactivations.size() == 1); // Only one element should be left (deactivations and
111  // activations were equal in number, but the head of
112  // activations was deleted at the beginning).
113  currentPhase.endTime = deactivations.begin()->first;
114  m_phases.push_back(currentPhase);
115 }
