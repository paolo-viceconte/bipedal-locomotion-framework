205  void setupOpti(const std::size_t& horizonSampling)
206  {
207  using Sl = casadi::Slice;
208 
209  // define the size of the optimization variables
210  this->optiVariables.dcm = this->opti.variable(this->dcmVectorSize, horizonSampling + 1);
211  this->optiVariables.omega = this->opti.variable(1, horizonSampling + 1);
212  this->optiVariables.vrp = this->opti.variable(this->vrpVectorSize, horizonSampling);
213  this->optiVariables.omegaDot = this->opti.variable(1, horizonSampling);
214 
215  // set the initial constraints
216  this->optiParameters.dcmInitialPosition = this->opti.parameter(this->dcmVectorSize);
217  this->optiParameters.dcmFinalPosition = this->opti.parameter(this->dcmVectorSize);
218  this->optiParameters.dcmInitialVelocity = this->opti.parameter(this->dcmVectorSize);
219  this->optiParameters.dcmFinalVelocity = this->opti.parameter(this->dcmVectorSize);
220  this->optiParameters.omegaInitialValue = this->opti.parameter();
221  this->optiParameters.omegaFinalValue = this->opti.parameter();
222 
223  // set constraints and cost functions whose equations does not change in the optimization
224  // problem
225  auto& dcm = this->optiVariables.dcm;
226  auto& vrp = this->optiVariables.vrp;
227  auto& omega = this->optiVariables.omega;
228  auto& omegaDot = this->optiVariables.omegaDot;
229 
230  // initial values
231  this->opti.subject_to(omega(Sl(), 0) == this->optiParameters.omegaInitialValue);
232  this->opti.subject_to(dcm(Sl(), 0) == this->optiParameters.dcmInitialPosition);
233 
234 
235  auto initialDcmVelocity = (omega(0) - omegaDot(0) / omega(0)) * (dcm(Sl(), 0) - vrp(Sl(), 0));
236  auto initialOmegaVelocity = omegaDot(0);
237 
238  this->opti.subject_to(casadi::MX::vertcat({initialDcmVelocity, initialOmegaVelocity})
239  == casadi::MX::vertcat({this->optiParameters.dcmInitialVelocity, 0}));
240 
241  // final values
242  this->opti.subject_to(omega(Sl(), -1) == this->optiParameters.omegaFinalValue);
243  this->opti.subject_to(dcm(Sl(), -1) == this->optiParameters.dcmFinalPosition);
244 
245  auto finalDcmVelocity = (omega(-1) - omegaDot(-1) / omega(-1)) * (dcm(Sl(), -1) - vrp(Sl(), -1));
246  auto finalOmegaVelocity = omegaDot(-1);
247 
248  this->opti.subject_to(casadi::MX::vertcat({finalDcmVelocity, finalOmegaVelocity})
249  == casadi::MX::vertcat({this->optiParameters.dcmFinalVelocity, 0}));
250 
251  auto trajectory = this->optiFunctions.dcmDynamics.map(horizonSampling);
252 
253  this->opti.subject_to(
254  casadi::MX::vertcat(
255  {dcm(Sl(), Sl(1, dcm.columns())), omega(Sl(), Sl(1, omega.columns()))})
256  == trajectory({casadi::MX::vertcat({dcm(Sl(), Sl(0, -1)), omega(Sl(), Sl(0, -1))}),
257  vrp,
258  omegaDot})[0]);
259 
260  // omega^2 - omegaDot has to be positive
261  this->opti.subject_to(casadi::MX::pow(omega(Sl(), Sl(0, -1)), 2) - omegaDot
262  > casadi::DM::zeros(omegaDot.rows(), omegaDot.columns()));
263 
264  // omega has to be positive
265  this->opti.subject_to(omega > casadi::DM::zeros(omega.rows(), omega.columns()));
266 
267  this->optiParameters.dcmRefererenceTraj = this->opti.parameter(this->dcmVectorSize, horizonSampling + 1);
268 
269  this->costFunction
270  = this->optiSettings.omegaDotWeight * casadi::MX::sumsqr(this->optiVariables.omegaDot)
271  + this->optiSettings.dcmTrackingWeight
272  * casadi::MX::sumsqr(this->optiVariables.dcm
273  - this->optiParameters.dcmRefererenceTraj)
274  + this->optiSettings.omegaDotRateOfChangeWeight
275  * casadi::MX::sumsqr(casadi::MX::diff(this->optiVariables.omegaDot))
276  + this->optiSettings.vrpRateOfChangeWeight
277  * casadi::MX::sumsqr(casadi::MX::diff(this->optiVariables.vrp.T()))
278  + this->optiSettings.dcmRateOfChangeWeight
279  * casadi::MX::sumsqr(casadi::MX::diff(this->optiVariables.dcm.T()));
280 
281  this->opti.minimize(costFunction);
282 
283  // set the opti object options
284  this->setupOptiOptions();
285  }

332  {

336  points.col(columnIndex) = point.head(numberOfCoordinates);
337  columnIndex++;

342  // TODO PLEASE OPTIMIZE ME

348  ecmpConstraintA = casadi::DM::zeros(A.rows(), A.cols());
349  std::memcpy(ecmpConstraintA.ptr(),
350  A.data(),

354  ecmpConstraintB.resize(b.size(), 1);

362  void computeVRPConstraint(const ContactPhaseList& contactPhaseList)
363  {
364  using Sl = casadi::Slice;
365 
366  casadi::DM ecmpConstraintA;
367  casadi::DM ecmpConstraintB;
368 
369  const auto& vrp = this->optiVariables.vrp;
370  const auto& omega = this->optiVariables.omega;
371  const auto& omegaDot = this->optiVariables.omegaDot;
372 
373  auto contactPhaseListIt = contactPhaseList.cbegin();
374  bool feetAreInSamePlane = computeConstraintElementsECMP(*contactPhaseListIt,
375  ecmpConstraintA,
376  ecmpConstraintB);
377 
378  for (std::size_t k = 0; k < this->optiVariables.vrp.columns(); k++)
379  {
380  // if the
381  if(k * this->optiSettings.plannerSamplingTime > contactPhaseListIt->endTime)
382  {
383  std::advance(contactPhaseListIt, 1);
384  feetAreInSamePlane = computeConstraintElementsECMP(*contactPhaseListIt,
385  ecmpConstraintA,
386  ecmpConstraintB);
387  }
388 
389  if (!feetAreInSamePlane)
390  {
391  this->opti.subject_to(
392  casadi::MX::mtimes(ecmpConstraintA,
393  casadi::MX::vertcat(this->optiFunctions.ecmp(
394  {omega(Sl(), k), vrp(Sl(), k), omegaDot(Sl(), k)})))
395  <= ecmpConstraintB);
396  } else
397  {
398  this->opti.subject_to(casadi::MX::mtimes(ecmpConstraintA, vrp(Sl(0, 2), k))
399  <= ecmpConstraintB);
400 
401  this->opti.subject_to(
402  vrp(2, k)
403  - this->optiSettings.gravity
404  / (casadi::MX::pow(omega(Sl(), k), 2) - omegaDot(Sl(), k))
405  == contactPhaseListIt->activeContacts.begin()->second->pose.translation()[2]);
406  }
407  }

410  bool computeDCMRegularization(const ContactPhaseList& contactPhaseList,
411  const DCMPlannerState& initialState)
412  {
413  std::vector<Eigen::VectorXd> dcmKnots;
414  std::vector<double> timeKnots;
415 
416  // first point
417  timeKnots.push_back(contactPhaseList.cbegin()->beginTime);
418  dcmKnots.push_back(initialState.dcmPosition);
419 
420  double averageDCMHeight = initialState.dcmPosition[2];
421  auto contactPhaseListIt = contactPhaseList.cbegin();
422 
423  for (std::size_t k = 0; k < this->optiVariables.vrp.columns(); k++)
424  {
425  // if the
426  if (k * this->optiSettings.plannerSamplingTime > contactPhaseListIt->endTime)
427  {
428  std::advance(contactPhaseListIt, 1);
429 
430  if (contactPhaseListIt->activeContacts.size() == 2
431  && contactPhaseListIt != contactPhaseList.begin()
432  && contactPhaseListIt != contactPhaseList.lastPhase())
433  {
434  timeKnots.emplace_back(
435  (contactPhaseListIt->endTime + contactPhaseListIt->beginTime) / 2);
436 
437  auto contactIt = contactPhaseListIt->activeContacts.cbegin();
438  const Eigen::Vector3d p1 = contactIt->second->pose.translation();
439  std::advance(contactIt, 1);
440  const Eigen::Vector3d p2 = contactIt->second->pose.translation();
441 
442  Eigen::Vector3d desiredDCMPosition = (p1 + p2) / 2.0;
443  desiredDCMPosition(2) += averageDCMHeight;
444 
445  dcmKnots.emplace_back(desiredDCMPosition);
446  }
447 
448  // TODO please try to make it more versatile
449  // read it as if the robot is in the last double support phase
450  else if (contactPhaseListIt->activeContacts.size() == 2
451  && contactPhaseListIt == contactPhaseList.lastPhase())
452  {
453  timeKnots.push_back(contactPhaseListIt->endTime);
454  auto contactIt = contactPhaseListIt->activeContacts.cbegin();
455  const Eigen::Vector3d p1 = contactIt->second->pose.translation();
456  std::advance(contactIt, 1);
457  const Eigen::Vector3d p2 = contactIt->second->pose.translation();
458 
459  Eigen::Vector3d desiredDCMPosition = (p1 + p2) / 2.0;
460  desiredDCMPosition(2) += averageDCMHeight;
461 
462  dcmKnots.emplace_back(desiredDCMPosition);
463  }
464  }
465  }
466  dcmRef.setInitialConditions(initialState.dcmVelocity, Eigen::Vector3d::Zero());
467  dcmRef.setFinalConditions(Eigen::Vector3d::Zero(), Eigen::Vector3d::Zero());
468  dcmRef.setKnots(dcmKnots, timeKnots);
469 
470  // populate the reference
471  this->initialValue.dcm = casadi::DM::zeros(3, this->optiVariables.dcm.columns());
472  Eigen::Vector3d velocity, acceleration;
473  Eigen::Map<Eigen::MatrixXd> initialValueDCMEigenMap(initialValue.dcm.ptr(),
474  initialValue.dcm.rows(),
475  initialValue.dcm.columns());
476 
477  for (int i = 0; i < this->optiVariables.dcm.columns(); i++)
478  {
479  const double currentTime = i * this->optiSettings.plannerSamplingTime;
480  if (!dcmRef.evaluatePoint(currentTime,
481  initialValueDCMEigenMap.col(i),
482  velocity,
483  acceleration))
484  {
485  log()->error("[TimeVaryingDCMPlanner::Impl::computeDCMRegularization] Unable to "
486  "evaluate the dcm reference trajectory.");
487  return false;
488  }
489  }
490  return true;
491  }

555  this->opti.set_initial(this->optiVariables.omega, this->initialValue.omega);
556  this->opti.set_initial(this->optiVariables.omegaDot,

710 bool TimeVaryingDCMPlanner::computeTrajectory()
711 {
712  assert(m_pimpl);
713 
714  constexpr auto logPrefix = "[TimeVaryingDCMPlanner::initialize]";
715 
716  if (!m_pimpl->isPlannerInitialized)
717  {
718  log()->error("{} Please initialize the planner before computing the trajectory.",
719  logPrefix);
720  return false;
721  }
722 
723  if (m_pimpl->isTrajectoryComputed)
724  {
725  log()->error("{} The trajectory has been already computed.", logPrefix);
726  return false;
727  }
728 
729  // clear the solver and the solution computed
730  m_pimpl->clear();
731 
732  const double& initialTrajectoryTime = m_contactPhaseList.cbegin()->beginTime;
733  const double& endTrajectoryTime = m_contactPhaseList.lastPhase()->endTime;
734 
735  m_pimpl->numberOfTrajectorySamples = std::ceil((endTrajectoryTime - initialTrajectoryTime)
736  / m_pimpl->optiSettings.plannerSamplingTime);
737 
738  m_pimpl->setupOpti(m_pimpl->numberOfTrajectorySamples);
739 
740  if (!m_pimpl->setupOptimizationProblem(m_contactPhaseList, m_initialState))
741  {
742  log()->error("{} Unable to setup the optimization problem.", logPrefix);
743  return false;
744  }
745 
746  // this is how casadi works
747  try
748  {
749  m_pimpl->optiSolution.solution = std::make_unique<casadi::OptiSol>(m_pimpl->opti.solve());
750  } catch (const std::exception& e)
751  {
752  log()->error("{} Unable to solve the optimization problem. The following exception has "
753  "been thrown by the solver: {}.",
754  logPrefix,
755  e.what());
756 
757  return false;
758  }
759 
760  m_pimpl->optiSolution.dcm = m_pimpl->optiSolution.solution->value(m_pimpl->optiVariables.dcm);
761  m_pimpl->optiSolution.vrp = m_pimpl->optiSolution.solution->value(m_pimpl->optiVariables.vrp);
762  m_pimpl->optiSolution.omega = m_pimpl->optiSolution.solution->value(m_pimpl->optiVariables.omega);
763  m_pimpl->optiSolution.omegaDot = m_pimpl->optiSolution.solution->value(m_pimpl->optiVariables.omegaDot);
764 
765  // reinitialize the trajectory
766  m_pimpl->trajectoryIndex = 0;
767 
768  m_pimpl->prepareSolution();
769 
770  m_pimpl->isTrajectoryComputed = true;
771 
772  return true;
773 }
