192 template <class PointType>
193 bool PointCloudProcessor<PointType>::downsample(const typename pcl::PointCloud<PointType>::Ptr inCloud,
194  typename pcl::PointCloud<PointType>::Ptr outCloud)
195 {
196  if (!checkInitialization())
197  {
198  return false;
199  }
200 
201  if (inCloud == nullptr)
202  {
203  log()->error("[PointCloudProcessor::downsample] Invalid pointer");
204  return false;
205  }
206 
207  if ((inCloud->size() / m_params.voxelSizeDownsampling[0]) >= std::numeric_limits<int>::max())
208  {
209  log()->warn("[PointCloudProcessor::downsample] "
210  " Input point cloud size too large for given voxel size x"
211  " This might cause overflow. Returning without any action.");
212  return false;
213  }
214 
215  if ((inCloud->size() / m_params.voxelSizeDownsampling[1]) >= std::numeric_limits<int>::max())
216  {
217  log()->warn("[PointCloudProcessor::downsample] "
218  " Input point cloud size too large for given voxel size y"
219  " This might cause overflow. Returning without any action.");
220  return false;
221  }
222 
223  if ((inCloud->size() / m_params.voxelSizeDownsampling[2]) >= std::numeric_limits<int>::max())
224  {
225  log()->warn("[PointCloudProcessor::downsample] "
226  " Input point cloud size too large for given voxel size z"
227  " This might cause overflow. Returning without any action.");
228  return false;
229  }
230 
231  if (inCloud->size() <= m_params.nrPointsForOutlierEstimation)
232  {
233  log()->error("[PointCloudProcessor::downsample] "
234  "Input point cloud size less than minimium required points for downsampling.");
235  return false;
236  }
237 
238  m_voxelGridFilter.setInputCloud(inCloud);
239  m_voxelGridFilter.filter(*outCloud);
240  return true;
241 }

270 template <class PointType>
271 bool PointCloudProcessor<PointType>::extractClusters(const typename pcl::PointCloud<PointType>::Ptr inCloud,
272  std::vector<pcl::PointIndices>& clusterIndices,
273  std::vector<typename pcl::PointCloud<PointType>::Ptr>& cloudClusters)
274 {
275  if (!checkInitialization())
276  {
277  return false;
278  }
279 
280  if (inCloud == nullptr)
281  {
282  log()->error("[PointCloudProcessor::extractClusters] Invalid pointer");
283  return false;
284  }
285 
286  if (inCloud->size() < m_params.minNrPointsInCluster)
287  {
288  log()->error("[PointCloudProcessor::extractClusters] "
289  "Input point cloud size less than minimium required points for clustering.");
290  return false;
291  }
292 
293  cloudClusters.clear();
294  clusterIndices.clear();
295  m_kdTree->setInputCloud(inCloud);
296  m_clusterExtractor.setSearchMethod(m_kdTree);
297  m_clusterExtractor.setInputCloud(inCloud);
298  m_clusterExtractor.extract(clusterIndices);
299 
300  for (auto it = clusterIndices.begin(); it != clusterIndices.end(); ++it)
301  {
302  auto clusteredCloud = pcl::make_shared<pcl::PointCloud<PointType>>();
303  for (auto pit = it->indices.begin(); pit != it->indices.end(); ++pit)
304  {
305  clusteredCloud->push_back ((*inCloud)[*pit]);
306  }
307  clusteredCloud->width = clusteredCloud->size();
308  clusteredCloud->height = 1;
309  clusteredCloud->is_dense = false;
310 
311  cloudClusters.emplace_back(clusteredCloud);
312  }
313  return true;
314 }
