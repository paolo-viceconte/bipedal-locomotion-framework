255  // run the thread
256  m_isRunning = true;
257  auto function = [&]() -> bool {
258  constexpr auto logPrefix = "[AdvanceableRunner::run]";
259  auto time = BipedalLocomotion::clock().now();
260  auto oldTime = time;
261  auto wakeUpTime = time;
262 
263  // the main periodic function
264  while (this->m_isRunning)
265  {
266  // detect if a clock has been reset
267  oldTime = time;
268  time = BipedalLocomotion::clock().now();
269  // if the current time is lower than old time, the timer has been reset.
270  if ((time - oldTime).count() < 1e-12)
271  {
272  wakeUpTime = time;
273  }
274 
275  // advance the wake-up time

277
